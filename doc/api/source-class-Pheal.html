 78:      * creates new Pheal API object
 79:      * @param int $userid the EVE userid/keyID

141:     /**
142:      * if userid+key is given it automatically detects (api call) the keyinfo and
143:      * set the correct access level for futher checks.
144:      *
145:      * Keep in mind this method will make an api request to account/APIKeyInfo based
146:      * on the given PhealConfig settings with the given key credentials.
147:      *
148:      * More important! This method will throw Exceptions on invalid keys or networks errors
149:      * So place this call into your try statement
150:      *
151:      * @throws PhealException|PhealAPIException|PhealHTTPException
152:      * @return bool|PhealResult
153:      */
154:     public function detectAccess()
155:     {
156:         // don't request keyinfo if api keys are not set or if new CAK aren't enabled
157:         if(!$this->userid || !$this->key || !PhealConfig::getInstance()->api_customkeys)
158:             return false;
159: 
160:         // request api key info, save old scope and restore it afterwords
161:         $old = $this->scope;
162:         $this->scope = "account";
163:         $keyinfo = $this->APIKeyInfo();
164:         $this->scope = $old;
165: 
166:         // set detected keytype and accessMask
167:         $this->setAccess($keyinfo->key->type, $keyinfo->key->accessMask);
168: 
169:         // return the APIKeyInfo Result object in the case you need it.
170:         return $keyinfo;
171:     }

173:     /**
174:      * method will ask caching class for valid xml, if non valid available
175:      * will make API call, and return the appropriate result
176:      * @throws PhealException|PhealAPIException|PhealHTTPException|PhealAccessException
177:      * @param string $scope api scope (examples: eve, map, server, ...)
178:      * @param string $name  api method (examples: ServerStatus, Kills, Sovereignty, ...)
179:      * @param array $opts   additional arguments (example: characterID => 12345, ...), should not contain apikey/userid/keyid/vcode
180:      * @return PhealResult
181:      */
182:     private function request_xml($scope, $name, array $opts = array())
183:     {
184:         $opts = array_merge(PhealConfig::getInstance()->additional_request_parameters, $opts);
185: 
186:         // apikey/userid/keyid|vcode shouldn't be allowed in arguments and removed to avoid wrong cached api calls
187:         foreach($opts AS $k => $v) {
188:             if(in_array(strtolower($k), array('userid','apikey','keyid','vcode')))
189:                 unset($opts[$k]);
190:         }
191: 
192:         // prepare http arguments + url (to not modify original argument list for cache saving)
193:         $url = PhealConfig::getInstance()->api_base . $scope . '/' . $name . ".xml.aspx";
194:         $use_customkey = (bool)PhealConfig::getInstance()->api_customkeys;
195:         $http_opts = $opts;
196:         if($this->userid) $http_opts[($use_customkey?'keyID':'userid')] = $this->userid;
197:         if($this->key) $http_opts[($use_customkey?'vCode':'apikey')] = $this->key;
198: 
199:         // check access level if given (throws PhealAccessExpception if API call is not allowed)
200:         if($use_customkey && $this->userid && $this->key && $this->keyType) {
201:             try {
202:                 PhealConfig::getInstance()->access->check($scope,$name,$this->keyType,$this->accessMask);
203:             } catch (Exception $e) {
204:                 PhealConfig::getInstance()->log->errorLog($scope,$name,$http_opts,$e->getMessage());
205:                 throw $e;
206:             }
207:         }
208: 
209:         // check cache first
210:         if(!$this->xml = PhealConfig::getInstance()->cache->load($this->userid,$this->key,$scope,$name,$opts))
211:         {
212:             try {
213:                 // start measure the response time
214:                 PhealConfig::getInstance()->log->start();
215: 
216:                 // request
217:                 if(PhealConfig::getInstance()->http_method == "curl" && function_exists('curl_init'))
218:                     $this->xml = self::request_http_curl($url,$http_opts);
219:                 else
220:                     $this->xml = self::request_http_file($url,$http_opts);
221: 
222:                 // stop measure the response time
223:                 PhealConfig::getInstance()->log->stop();
224: 
225:                 // parse
226:                 $element = new SimpleXMLElement($this->xml);
227: 
228:             // just forward HTTP Errors
229:             } catch(PhealHTTPException $e) {
230:                 throw $e;
231:             // just forward PhealConnectionException errors
232:             } catch(PhealConnectionException $e) {
233:                 throw $e;
234:             // other request errors
235:             } catch(Exception $e) {
236:                 // log + throw error
237:                 PhealConfig::getInstance()->log->errorLog($scope,$name,$http_opts,$e->getCode() . ': ' . $e->getMessage());
238:                 // change Exception to PhealException but keep the original error data (easier debugging in client application).
239:                 throw new PhealException('Original exception: ' . $e->getMessage(), $e->getCode(), $e);
240:             }
241:             PhealConfig::getInstance()->cache->save($this->userid,$this->key,$scope,$name,$opts,$this->xml);
242:             
243:             // archive+save only non-error api calls + logging
244:             if(!$element->error) {
245:                 PhealConfig::getInstance()->log->log($scope,$name,$http_opts);
246:                 PhealConfig::getInstance()->archive->save($this->userid,$this->key,$scope,$name,$opts,$this->xml);
247:             } else {
248:                 PhealConfig::getInstance()->log->errorLog($scope,$name,$http_opts,$element->error['code'] . ': ' . $element->error);
249:             }
250:         } else {
251:             $element = new SimpleXMLElement($this->xml);
252:         }
253:         return new PhealResult($element);
254:     }

256:     /**
257:      * method will do the actual http call using curl libary. 
258:      * you can choose between POST/GET via config.
259:      * will throw Exception if http request/curl times out or fails
260:      * @param String $url url beeing requested
261:      * @param array $opts an array of query paramters
262:      * @return string raw http response
263:      */
264:     public static function request_http_curl($url,$opts)
265:     {
266:         // init curl
267:         if(!(is_resource(self::$curl) && get_resource_type(self::$curl) == 'curl'))
268:             self::$curl = curl_init();
269: 
270:         // custom user agent
271:         if(($http_user_agent = PhealConfig::getInstance()->http_user_agent) != false)
272:             curl_setopt(self::$curl, CURLOPT_USERAGENT, $http_user_agent);
273:         
274:         // custom outgoing ip address
275:         if(($http_interface_ip = PhealConfig::getInstance()->http_interface_ip) != false)
276:             curl_setopt(self::$curl, CURLOPT_INTERFACE, $http_interface_ip);
277: 
278:         // ignore ssl peer verification if needed
279:         if(substr($url,0,5) == "https")
280:             curl_setopt(self::$curl, CURLOPT_SSL_VERIFYPEER, PhealConfig::getInstance()->http_ssl_verifypeer);
281:             
282:         // http timeout 
283:         if(($http_timeout = PhealConfig::getInstance()->http_timeout) != false)
284:             curl_setopt(self::$curl, CURLOPT_TIMEOUT, $http_timeout);
285:             
286:         // use post for params
287:         if(count($opts) && PhealConfig::getInstance()->http_post)
288:         {
289:             curl_setopt(self::$curl, CURLOPT_POST, true);
290:             curl_setopt(self::$curl, CURLOPT_POSTFIELDS, $opts);
291:         }
292:         else
293:         {
294:             curl_setopt(self::$curl, CURLOPT_POST, false);
295:             
296:             // attach url parameters
297:             if(count($opts))
298:                 $url .= "?" . http_build_query($opts,'', '&');
299:         }
300:         
301:         // additional headers
302:         $headers = array();
303:         
304:         // enable/disable keepalive
305:         if(($http_keepalive = PhealConfig::getInstance()->http_keepalive) != false)
306:         {
307:             curl_setopt(self::$curl, CURLOPT_FORBID_REUSE, false);
308:             $http_keepalive = ($http_keepalive === true) ? 15 : (int)$http_keepalive;
309:             $headers[] = "Connection: keep-alive";
310:             $headers[] = "Keep-Alive: timeout=" . $http_keepalive . ", max=1000";
311:         }
312:         else 
313:         {
314:             curl_setopt(self::$curl, CURLOPT_FORBID_REUSE, true);
315:         }
316: 
317:         // allow all encodings
318:         curl_setopt(self::$curl, CURLOPT_ENCODING, "");
319: 
320:         // curl defaults
321:         curl_setopt(self::$curl, CURLOPT_URL, $url);
322:         curl_setopt(self::$curl, CURLOPT_HTTPHEADER, $headers);
323:         curl_setopt(self::$curl, CURLOPT_RETURNTRANSFER, true);
324:         
325:         // call
326:         $result = curl_exec(self::$curl);
327:         $errno = curl_errno(self::$curl);
328:         $error = curl_error(self::$curl);
329: 
330:         // response http headers
331:         $httpCode = curl_getinfo(self::$curl, CURLINFO_HTTP_CODE);
332: 
333:         if(!PhealConfig::getInstance()->http_keepalive)
334:             self::disconnect();
335: 
336:         // http errors
337:         if($httpCode >= 400)
338:             throw new PhealHTTPException($httpCode, $url);
339: 
340:         // curl errors
341:         if($errno)
342:             throw new PhealConnectionException($error, $errno);
343:         else
344:             return $result;
345:     }

347:     /**
348:      * method will do the actual http call using file()
349:      * remember: on some installations, file_get_contents(url) might not be available due to
350:      * restrictions via allow_url_fopen
351:      * @param String $url url beeing requested
352:      * @param array $opts an array of query paramters
353:      * @return string raw http response
354:      */
355:     public static function request_http_file($url,$opts)
356:     {
357:         $options = array();
358:         
359:         // set custom user agent
360:         if(($http_user_agent = PhealConfig::getInstance()->http_user_agent) != false)
361:             $options['http']['user_agent'] = $http_user_agent;
362:         
363:         // set custom http timeout
364:         if(($http_timeout = PhealConfig::getInstance()->http_timeout) != false)
365:             $options['http']['timeout'] = $http_timeout;
366:         
367:         // ignore ssl peer verification if needed
368:         if(substr($url,5) == "https")
369:             $options['ssl']['verify_peer'] = PhealConfig::getInstance()->http_ssl_verifypeer;
370:         
371:         // use post for params
372:         if(count($opts) && PhealConfig::getInstance()->http_post)
373:         {
374:             $options['http']['method'] = 'POST';
375:             $options['http']['content'] = http_build_query($opts, '', '&');
376:         }
377:         // else build url parameters
378:         elseif(count($opts))
379:         {
380:             $url .= "?" . http_build_query($opts,'','&');
381:         }
382: 
383:         // set track errors. needed for $php_errormsg
384:         $oldTrackErrors = ini_get('track_errors');
385:         ini_set('track_errors', true);
386: 
387:         // create context with options and request api call
388:         // suppress the 'warning' message which we'll catch later with $php_errormsg
389:         if(count($options)) 
390:         {
391:             $context = stream_context_create($options);
392:             $result = @file_get_contents($url, false, $context);
393:         } else {
394:             $result = @file_get_contents($url);
395:         }
396:         
397:         // check for http errors via magic $http_response_header
398:         $httpCode = 200;
399:         if(isset($http_response_header[0]))
400:             list($httpVersion,$httpCode,$httpMsg) = explode(' ', $http_response_header[0], 3);
401:         
402:         // throw http error
403:         if(is_numeric($httpCode) && $httpCode >= 400)
404:             throw new PhealHTTPException($httpCode, $url);
405: 
406:         // throw error
407:         if($result === false) {
408:             $message = ($php_errormsg ? $php_errormsg : 'HTTP Request Failed');
409:             
410:             // set track_errors back to the old value
411:             ini_set('track_errors',$oldTrackErrors);
412: 
413:             throw new PhealConnectionException($message);
414: 
415:         // return result
416:         } else {
417:             // set track_errors back to the old value
418:             ini_set('track_errors',$oldTrackErrors);
419:             return $result;
420:         }
421:     }
