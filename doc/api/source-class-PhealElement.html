 28: /**
 29:  * PhealElement holds elements of the EVE API
 30:  */
 31: class PhealElement implements PhealArrayInterface
 32: {
 33:     /**
 34:      * Name of the Element
 35:      * @var String
 36:      */
 37:     public $_name;
 38: 
 39:     /**
 40:      * Value of the Element
 41:      * @var mixed
 42:      */
 43:     public $_value;
 44:     
 45:     /**
 46:      * container containing information that the EVE API stored in XML attributes
 47:      * @var array
 48:      */
 49:     public $_attribs = array();
 50: 
 51:     /**
 52:      * create new PhealElement
 53:      * @param string $name
 54:      * @param mixed $value
 55:      */
 56:     protected function  __construct($name, $value)
 57:     {
 58:         $this->_name = $name;
 59:         $this->_value = $value;
 60:     }
 61: 
 62:     /**
 63:      * Add an attribute to the attributes container
 64:      * @param string $key
 65:      * @param string $val
 66:      */
 67:     public function add_attrib($key, $val)
 68:     {
 69:         $this->_attribs = array_merge(array($key => $val), $this->_attribs);
 70:     }
 71: 
 72:     /**
 73:      * Magic method, will check if name is in attributes, if not pass
 74:      * the request on to what ever is stored in value
 75:      * @param string $name
 76:      * @return mixed
 77:      */
 78:     public function __get($name)
 79:     {
 80:         if(isset($this->_attribs[$name]))
 81:             return $this->_attribs[$name];
 82:         return $this->_value->$name;
 83:     }
 84: 
 85:     /**
 86:      * returns the Object as associated array
 87:      * @return array
 88:      */
 89:     public function toArray()
 90:     {
 91:         $return = array();
 92:         foreach($this->_attribs AS $key => $value)
 93:             $return[$key] = $value;
 94: 
 95:         if($this->_value instanceof PhealArrayInterface)
 96:             $return = array_merge($return, $this->_value->toArray());
 97:         else
 98:             $return[$this->_name] = $this->_value;
 99: 
100:         return $return;
101:     }
102: 
103:     /**
104:      * parse SimpleXMLElement
105:      * @param SimpleXMLElement $element
106:      * @return mixed
107:      */
108:     public static function parse_element($element)
109:     {
110:         if($element->getName() =="rowset")
111:         {
112:             $re = new PhealRowSet($element);
113:         }
114:         // corp/MemberSecurity workaround
115:         elseif($element->getName() == "result" && $element->member)
116:         {
117:             $container = new PhealContainer();
118:             $container->add_element('members',new PhealRowSet($element,'members','member'));
119:             $re = new PhealElement('result',$container);
120:         }
121:         else
122:         {
123:             $key = $element->getName();
124:             $echilds = $element->children();
125:             if(count($echilds) > 0)
126:             {
127:                 $container = new PhealContainer();
128:                 foreach($echilds as $celement)
129:                 {
130:                     $cel = PhealElement::parse_element($celement);
131:                     if(count($celement->attributes()) > 0)
132:                         $container->add_element($cel->_name, $cel);
133:                     else
134:                         $container->add_element($cel->_name, $cel->_value);
135:                 }
136:                 $value = $container;
137:             } else $value = (String) $element;
138: 
139:             $re = new PhealElement($key, $value);
140:             if(count($element->attributes()) > 0)
141:             {
142:                 foreach($element->attributes() as $attelem)
143:                 {
144:                     $re->add_attrib($attelem->getName(), (String) $attelem);
145:                 }
146:             }
147: 
148:         }
149:         return $re;
150:     }
151: }
