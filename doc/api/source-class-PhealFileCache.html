 27: /**
 28:  * Simple filecache for the xml
 29:  */
 30: class PhealFileCache implements PhealCacheInterface
 31: {
 32:     /**
 33:      * path where to store the xml
 34:      * @var string
 35:      */
 36:     protected $basepath;
 37: 
 38:     /**
 39:      * various options for the filecache
 40:      * valid keys are: delimiter, umask, umask_directory
 41:      * @var array
 42:      */
 43:     protected $options = array(
 44:         'delimiter' => ':',
 45:         'umask' => 0666,
 46:         'umask_directory' => 0777
 47:     );
 48: 
 49:     /**
 50:      * construct PhealFileCache,
 51:      * @param string $basepath optional string on where to store files, defaults to the current/users/home/.pheal/cache/
 52:      * @param array $options optional config array, valid keys are: delimiter, umask, umask_directory
 53:      */
 54:     public function __construct($basepath = false, $options = array())
 55:     {
 56:         if(!$basepath)
 57:             $basepath = getenv('HOME'). "/.pheal/cache/";
 58:         $this->basepath = $basepath;
 59: 
 60:         // Windows systems don't allow : as part of the filename
 61:         $this->options['delimiter'] = (strtoupper (substr(PHP_OS, 0,3)) == 'WIN') ? "#" : ":";
 62: 
 63:         // add options
 64:         if(is_array($options) && count($options))
 65:             $this->options = array_merge($this->options, $options);
 66:     }
 67: 
 68:     /**
 69:      * create a filename to use
 70:      * @param int $userid
 71:      * @param string $apikey
 72:      * @param string $scope
 73:      * @param string $name
 74:      * @param array $args
 75:      * @return string
 76:      */
 77:     protected function filename($userid, $apikey, $scope, $name, $args)
 78:     {
 79:         // secure input to make sure pheal don't write the files anywhere
 80:         // user can define their own apikey/vcode
 81:         // maybe this should be tweaked or hashed
 82:         $regexp = "/[^a-z0-9,.-_=]/i";
 83:         $userid = (int)$userid;
 84:         $apikey = preg_replace($regexp,'_',$apikey);
 85:         
 86:         // build cache filename
 87:         $argstr = "";
 88:         foreach($args as $key => $val) {
 89:             if(strlen($val) < 1)
 90:                 unset($args[$key]);
 91:             elseif(!in_array(strtolower($key), array('userid','apikey','keyid','vcode')))
 92:                 $argstr .= preg_replace($regexp,'_',$key) . $this->options['delimiter'] . preg_replace($regexp,'_',$val) . $this->options['delimiter'];
 93:         }
 94:         $argstr = substr($argstr, 0, -1);
 95:         $filename = "Request" . ($argstr ? "_" . $argstr : "") . ".xml";
 96:         $filepath = $this->basepath . ($userid ? "$userid/$apikey/$scope/$name/" : "public/public/$scope/$name/");
 97:         
 98:         if(!file_exists($filepath)) {
 99:             // check write access
100:             if(!is_writable($this->basepath))
101:                 throw new PhealException(sprintf("Cache directory '%s' isn't writeable", $filepath));
102: 
103:             // create cache folder
104:             $oldUmask = umask(0);
105:             mkdir($filepath, $this->options['umask_directory'], true);
106:             umask($oldUmask);
107: 
108:         } else {
109:             // check write access
110:             if(!is_writable($filepath))
111:                 throw new PhealException(sprintf("Cache directory '%s' isn't writeable", $filepath));
112:             if(file_exists($filename) && !is_writeable($filename))
113:                 throw new PhealException(sprintf("Cache file '%s' isn't writeable", $filename));
114:         }
115:         return $filepath . $filename;
116:     }
117: 
118:     /**
119:      * Load XML from cache
120:      * @param int $userid
121:      * @param string $apikey
122:      * @param string $scope
123:      * @param string $name
124:      * @param array $args
125:      */
126:     public function load($userid, $apikey, $scope, $name, $args)
127:     {
128:         $filename = $this->filename($userid, $apikey, $scope, $name, $args);
129:         if(!file_exists($filename))
130:             return false;
131:         $xml = file_get_contents($filename);
132:         if($this->validate_cache($xml))
133:             return $xml;
134:         return false;
135: 
136:     }
137: 
138:     /**
139:      * validate the cached xml if it is still valid. This contains a name hack
140:      * to work arround EVE API giving wrong cachedUntil values
141:      * @param string $xml
142:      * @return boolean
143:      */
144:     public function validate_cache($xml)
145:     {
146:         $tz = date_default_timezone_get();
147:         date_default_timezone_set("UTC");
148: 
149:         $xml = new SimpleXMLElement($xml);
150:         $dt = (int) strtotime($xml->cachedUntil);
151:         $time = time();
152: 
153:         date_default_timezone_set($tz);
154: 
155:         return (bool) ($dt > $time);
156:     }
157:     
158:     /**
159:      * Save XML from cache
160:      * @param int $userid
161:      * @param string $apikey
162:      * @param string $scope
163:      * @param string $name
164:      * @param array $args
165:      * @param string $xml
166:      */
167:     public function save($userid,$apikey,$scope,$name,$args,$xml) 
168:     {
169:         $filename = $this->filename($userid, $apikey, $scope, $name, $args);
170:         $exists = file_exists($filename);
171:         
172:         // save content
173:         file_put_contents($filename, $xml);
174:         
175:         // chmod only new files
176:         if(!$exists)
177:             chmod($filename, $this->options['umask']);
178:     }
179: }
